42static int _set_blocking (int fd, int should_block)
43{
44 struct termios tty;
45 memset (&tty, 0, sizeof tty);
46 if (tcgetattr (fd, &tty) != 0)
47 {
48 return errno;
49 }
50
51 tty.c_cc[VMIN] = should_block ? 1 : 0;
52 tty.c_cc[VTIME] = 0; // 0.5 seconds read timeout
53
54 if (tcsetattr (fd, TCSANOW, &tty) != 0)
55 return errno;
56 return 0;
57}

62static int _set_interface_attribs (int fd, int speed, int parity)
63{
64 struct termios tty;
65 if (tcgetattr (fd, &tty) != 0)
66 {
67 // error_message ("error %d from tcgetattr", errno);
68 return -1;
69 }
70
71 cfsetospeed (&tty, speed);
72 cfsetispeed (&tty, speed);
73 tty.c_cflag = (tty.c_cflag & ~CSIZE) | CS8; // 8-bit chars
74 // disable IGNBRK for mismatched speed tests; otherwise receive break
75 // as \000 chars
76 tty.c_iflag &= ~IGNBRK; // disable break processing
77 tty.c_lflag = 0; // no signaling chars, no echo,
78 // no canonical processing
79 tty.c_oflag = 0; // no remapping, no delays
80 tty.c_cc[VMIN] = 0; // read doesn't block
81 tty.c_cc[VTIME] = 5; // 0.5 seconds read timeout
82
83 tty.c_iflag &= ~(IGNBRK | BRKINT | PARMRK | ISTRIP | INLCR | IGNCR | ICRNL | IXON); //tty is the name of the struct termios
84
85 tty.c_cflag |= (CLOCAL | CREAD);// ignore modem controls,
86 // enable reading
87 tty.c_cflag &= ~(PARENB | PARODD); // shut off parity
88 tty.c_cflag |= parity;
89 tty.c_cflag &= ~CSTOPB;
90 tty.c_cflag &= ~CRTSCTS;
91 if (tcsetattr (fd, TCSANOW, &tty) != 0)
92 {
93 return errno;
94 }
95 return 0;
96}
