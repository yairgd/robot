46 void StateTrack(unsigned char c) override {
47 switch (state) {
48 case ReceiveState::Preamble:
49 if (c == 0xaa) {
50 state = MsgType;
51 msg.crc = 0xaa;
52 idx = 0;
53 state = ReceiveState::MsgType;
54 } else {
55 state = ReceiveState::Preamble;
56 }
57 break;
58 case ReceiveState::MsgType:
59 //msg.type = 125;//static_cast<Simple::MsgType>(c);
60 msg.crc+=c;
61 state = ReceiveState::Cmd;
62 break;
63 case ReceiveState::Cmd:
64 msg.cmd =static_cast<CmdId>(c);
65 msg.crc+=c;
66 state = ReceiveState::Length;
67 length = 0;
68 break;
69 case ReceiveState::Length:
70 msg.crc+=c;
71 length = c - 1;
72 msg.data.size = c;
73 state = ReceiveState::Data;
74 break;
75 case ReceiveState::Data:
76 msg.crc+=c;
77 msg.data.buffer[idx++] = c;
78 // update crc with c
79 if (length) {
80 length--;
81 } else {
82 state = ReceiveState::CheckSum;
83 }
84 break;
85 case ReceiveState::CheckSum:
86 msg.crc+=0xee;
87 if (c != msg.crc ) {
88 state = ReceiveState::Preamble;
89 m_handleMsg->Wrongdata();
90 } else{
91 state = ReceiveState::Suffix;
92 }
93 break;
94 case ReceiveState::Suffix:
95 if (c != 0xee) {
96 m_handleMsg->Wrongdata();
97 } else {
98 m_handleMsg->NewData(msg);
99
100 }
101 state = ReceiveState::Preamble;
102 break;
103 }
